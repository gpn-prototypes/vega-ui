```tsx
import { ScalePanel } from '@gpn-prototypes/vega-ui';

export const MyComponent = () => {
  const [scale, setScale] = useState(100);

  const handleChange = (newScale) => {
    setScale(newScale);
  };

  const handleAlign = () => {
    console.log('align');
  };

  return (
    <ScalePanel
      step={10}
      scale={scale}
      minScale={20}
      maxScale={150}
      onChange={handleChange}
      onAlign={handleAlign}
    />
  );
};
```
